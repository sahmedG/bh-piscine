package piscine

import "github.com/01-edu/z01"

func PrintComb() {
	for i := '0'; i < '8'; i++ {
		for j := '1'; j < '9'; j++ {
			if j > i {
				for k := '2'; k <= '9'; k++ {
					if k > j {

						z01.PrintRune(i)
						z01.PrintRune(j)
						z01.PrintRune(k)
						if i == '7' && j == '8' && k == '9' {
							break
						}
						z01.PrintRune(',')
						z01.PrintRune(' ')
					}
				}
			}
		}
	}
	z01.PrintRune('\n')
}
